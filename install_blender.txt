# I chose to install in the parent directory
cd ..
mkdir blender 
cd blender
wget https://download.blender.org/release/Blender5.0/blender-5.0.1-linux-x64.tar.xz
tar -xf blender-5.0.1-linux-x64.tar.xz
# Adapt this path to your case !
echo 'export PATH=/home/k.kassab/panorama/blender/blender-5.0.1-linux-x64:$PATH' >> ~/.bashrc
echo 'export PATH=/home/a.schnepf/blender/blender-5.0.1-linux-x64:$PATH' >> ~/.bashrc


# Install some packages into blender's python interpreter
## 1. get your blender python path using 
blender --background --python-expr "import sys; print(sys.executable)"

## 2. Paste the path given into BLENDER_PY below TODO !
BLENDER_PY="/home/k.kassab/panorama/blender/blender-5.0.1-linux-x64/5.0/python/bin/python3.11"
BLENDER_PY="/home/a.schnepf/blender/blender-5.0.1-linux-x64/5.0/python/bin/python3.11"

## 3. Make sure blender's python interpreter has the latest pip
$BLENDER_PY -m ensurepip --upgrade
$BLENDER_PY -m pip install --upgrade pip

## 4. Install the following libraries into blender's python interpreter
$BLENDER_PY -m pip install prodict
$BLENDER_PY -m pip install pyfiglet
$BLENDER_PY -m pip install PyYAML
$BLENDER_PY -m pip install pathlib
$BLENDER_PY -m pip install numpy
$BLENDER_PY -m pip install opencv-python
/!\ if you need to install packages from blender, you should install it using the blender's python 
interpreter, not the normal pip install !
